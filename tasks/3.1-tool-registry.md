### 1. Create Tool Registry Core

**File:** `src/services/toolRegistry.ts`

```typescript
import { v4 as uuid } from 'uuid';
import type { IntelligenceEvent } from '@/contracts';

export interface ToolDefinition {
  name: string;
  description: string;
  category: 'search' | 'scan' | 'enrich' | 'analyze';
  executor: (params: any) => Promise<any>;  // Raw tool execution
  normalizer?: (raw: any, params: any) => IntelligenceEvent;  // Convert to canonical schema
  schema: any;                               // Input schema (Zod)
  cost: number;                              // Estimated cost (1-10 scale)
  reliability: number;                       // Historical reliability (0.0-1.0)
  rateLimit?: {
    requests: number;
    window: number;  // milliseconds
  };
  metadata: {
    version: string;
    author: string;
    requiresAuth: boolean;
    tags: string[];
  };
}

export class ToolRegistry {
  private tools: Map<string, ToolDefinition> = new Map();
  private executionHistory: Map<string, number> = new Map();

  register(tool: ToolDefinition): void {
    this.tools.set(tool.name, tool);
  }

  get(name: string): ToolDefinition | undefined {
    return this.tools.get(name);
  }

  list(category?: string): ToolDefinition[] {
    const allTools = Array.from(this.tools.values());
    if (category) {
      return allTools.filter(t => t.category === category);
    }
    return allTools;
  }

  async execute(
    name: string,
    params: any,
    options?: { normalize?: boolean; traceId?: string }
  ): Promise<any> {
    const tool = this.get(name);
    if (!tool) {
      throw new Error(`Tool not found: ${name}`);
    }

    // Rate limiting check
    if (tool.rateLimit) {
      this.checkRateLimit(name, tool.rateLimit);
    }

    const startTime = Date.now();
    try {
      // Execute tool
      const rawResult = await tool.executor(params);

      // Track execution
      this.trackExecution(name, Date.now() - startTime, true);

      // Normalize if requested
      if (options?.normalize && tool.normalizer) {
        return tool.normalizer(rawResult, params);
      }

      return rawResult;
    } catch (error: any) {
      this.trackExecution(name, Date.now() - startTime, false);
      throw error;
    }
  }

  async executeAndNormalize(
    name: string,
    params: any,
    target: string,
    traceId: string
  ): Promise<IntelligenceEvent> {
    const tool = this.get(name);
    if (!tool) {
      throw new Error(`Tool not found: ${name}`);
    }

    if (!tool.normalizer) {
      throw new Error(`Tool ${name} does not have a normalizer`);
    }

    const startTime = Date.now();
    try {
      const rawResult = await tool.executor(params);
      const event = tool.normalizer(rawResult, params);

      // Enrich event with metadata
      return {
        ...event,
        id: uuid(),
        traceId,
        tool: name,
        target,
        timestamp: new Date(),
        metadata: {
          ...event.metadata,
          executionTime: Date.now() - startTime,
          cost: tool.cost
        }
      };
    } catch (error: any) {
      // Return failed event
      return {
        id: uuid(),
        traceId,
        tool: name,
        target,
        timestamp: new Date(),
        status: 'failed',
        entities: [],
        observations: [],
        relationships: [],
        metadata: {
          executionTime: Date.now() - startTime,
          cost: tool.cost,
          errors: [error.message]
        }
      };
    }
  }

  private checkRateLimit(toolName: string, limit: { requests: number; window: number }): void {
    // Simple in-memory rate limiting
    const key = `${toolName}:${Math.floor(Date.now() / limit.window)}`;
    const count = this.executionHistory.get(key) || 0;

    if (count >= limit.requests) {
      throw new Error(`Rate limit exceeded for ${toolName}`);
    }

    this.executionHistory.set(key, count + 1);
  }

  private trackExecution(toolName: string, duration: number, success: boolean): void {
    // Update reliability score
    const tool = this.get(toolName);
    if (tool) {
      const alpha = 0.1; // Exponential moving average factor
      const outcome = success ? 1.0 : 0.0;
      tool.reliability = alpha * outcome + (1 - alpha) * tool.reliability;
    }
  }
}

// Singleton instance
export const toolRegistry = new ToolRegistry();
```

#### b. DNS Lookup Tool
**File:** `src/services/tools/dnsLookup.ts`

```typescript
import { z } from 'zod';
import { promises as dns } from 'dns';
import type { ToolDefinition } from '../toolRegistry';
import type { IntelligenceEvent, EntityReference, Observation } from '@/contracts';

const schema = z.object({
  domain: z.string()
});

const executor = async (params: z.infer<typeof schema>) => {
  try {
    const [addresses, mx] = await Promise.all([
      dns.resolve4(params.domain).catch(() => []),
      dns.resolveMx(params.domain).catch(() => [])
    ]);

    return {
      domain: params.domain,
      addresses,
      mx,
      timestamp: new Date()
    };
  } catch (error: any) {
    throw new Error(`DNS lookup failed: ${error.message}`);
  }
};

const normalizer = (raw: any, params: any): IntelligenceEvent => {
  const entities: EntityReference[] = [{
    id: params.domain,
    type: 'infrastructure',
    name: params.domain,
    aliases: [],
    confidence: 1.0,
    attributes: {
      ipAddresses: raw.addresses,
      mxRecords: raw.mx
    },
    sourceIds: [],
    firstSeen: new Date(),
    lastSeen: new Date(),
    metadata: {
      verified: true,
      tags: ['dns']
    }
  }];

  const observations: Observation[] = [
    ...raw.addresses.map((ip: string) => ({
      id: `obs-${ip}`,
      entityId: params.domain,
      type: 'attribute' as const,
      property: 'ip_address',
      value: ip,
      confidence: 1.0,
      source: {
        tool: 'dns_lookup',
        timestamp: raw.timestamp,
        raw: raw
      },
      context: {}
    })),
    ...raw.mx.map((mx: any) => ({
      id: `obs-mx-${mx.exchange}`,
      entityId: params.domain,
      type: 'attribute' as const,
      property: 'mx_record',
      value: mx.exchange,
      confidence: 1.0,
      source: {
        tool: 'dns_lookup',
        timestamp: raw.timestamp,
        raw: raw
      },
      context: { priority: mx.priority }
    }))
  ];

  return {
    id: '',
    traceId: '',
    tool: 'dns_lookup',
    target: params.domain,
    timestamp: raw.timestamp,
    status: 'success',
    entities,
    observations,
    relationships: [],
    metadata: {
      executionTime: 0,
      raw
    }
  };
};

export const dnsLookupTool: ToolDefinition = {
  name: 'dns_lookup',
  description: 'Perform DNS lookups for a domain',
  category: 'scan',
  executor,
  normalizer,
  schema,
  cost: 1,
  reliability: 0.99,
  metadata: {
    version: '1.0.0',
    author: 'shaivra',
    requiresAuth: false,
    tags: ['dns', 'infrastructure', 'network']
  }
};
```

```typescript
import { describe, it, expect, beforeEach } from 'vitest';
import { ToolRegistry } from '@/services/toolRegistry';
import { lookupEntityTool } from '@/services/tools/lookupEntity';

describe('ToolRegistry', () => {
  let registry: ToolRegistry;

  beforeEach(() => {
    registry = new ToolRegistry();
  });

  describe('Tool Registration', () => {
    it('registers a tool successfully', () => {
      registry.register(lookupEntityTool);

      const tool = registry.get('lookup_entity');
      expect(tool).toBeDefined();
      expect(tool?.name).toBe('lookup_entity');
    });

    it('lists all registered tools', () => {
      registry.register(lookupEntityTool);

      const tools = registry.list();
      expect(tools.length).toBe(1);
    });

    it('filters tools by category', () => {
      registry.register(lookupEntityTool);

      const searchTools = registry.list('search');
      expect(searchTools.length).toBe(1);
      expect(searchTools[0].category).toBe('search');
    });
  });

  describe('Tool Execution', () => {
    beforeEach(() => {
      registry.register(lookupEntityTool);
    });

    it('executes tool successfully', async () => {
      const result = await registry.execute('lookup_entity', {
        entityId: 'test-123',
        entityType: 'person'
      });

      expect(result).toBeDefined();
    });

    it('throws error for non-existent tool', async () => {
      await expect(
        registry.execute('non_existent', {})
      ).rejects.toThrow('Tool not found');
    });

    it('executes and normalizes in one call', async () => {
      const event = await registry.executeAndNormalize(
        'lookup_entity',
        { entityId: 'test-123', entityType: 'person' },
        'test-target',
        'trace-123'
      );

      expect(event.id).toBeDefined();
      expect(event.traceId).toBe('trace-123');
      expect(event.tool).toBe('lookup_entity');
      expect(event.entities).toBeDefined();
    });

    it('handles execution failures gracefully', async () => {
      // Mock a failing tool
      const failingTool = {
        ...lookupEntityTool,
        name: 'failing_tool',
        executor: async () => {
          throw new Error('Tool execution failed');
        }
      };

      registry.register(failingTool);

      const event = await registry.executeAndNormalize(
        'failing_tool',
        {},
        'test-target',
        'trace-123'
      );

      expect(event.status).toBe('failed');
      expect(event.metadata.errors).toBeDefined();
    });
  });

  describe('Rate Limiting', () => {
    it('enforces rate limits', async () => {
      const limitedTool = {
        ...lookupEntityTool,
        name: 'limited_tool',
        rateLimit: {
          requests: 2,
          window: 1000
        }
      };

      registry.register(limitedTool);

      // First two requests should succeed
      await registry.execute('limited_tool', { entityId: '1', entityType: 'person' });
      await registry.execute('limited_tool', { entityId: '2', entityType: 'person' });

      // Third request should fail
      await expect(
        registry.execute('limited_tool', { entityId: '3', entityType: 'person' })
      ).rejects.toThrow('Rate limit exceeded');
    });
  });

  describe('Reliability Tracking', () => {
    it('updates reliability score based on success', async () => {
      registry.register(lookupEntityTool);

      const initialReliability = lookupEntityTool.reliability;

      // Execute successfully
      await registry.execute('lookup_entity', {
        entityId: 'test',
        entityType: 'person'
      });

      const tool = registry.get('lookup_entity');
      // Reliability should remain high for successful execution
      expect(tool?.reliability).toBeGreaterThanOrEqual(0.9);
    });
  });
});
```
